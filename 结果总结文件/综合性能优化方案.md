### 2.1 翻译结果LRU缓存 (P0 - 高优先级)

#### 问题分析
- 翻译函数直接查询数据库，无缓存机制
- 相同文本重复翻译，浪费资源
- 数据库查询开销大

#### 实现方案

**文件位置**: `app/server/translate/local_translate.py`

```python
from functools import lru_cache
from ..dao.dao import fetch_one

# 添加LRU缓存装饰器
@lru_cache(maxsize=500)
def _get_translation_cache(word: str) -> tuple | None:
    """缓存翻译结果的内部函数"""
    # 先查tag_tags表
    query = "SELECT desc, color FROM tag_tags WHERE text = ?"
    result = await fetch_one("tags", query, (word,))
    if result:
        return ("tag_tags", result[0], result[1])
    
    # 再查danbooru_tag表
    query = "SELECT translate, color_id FROM danbooru_tag WHERE tag = ?"
    result = await fetch_one("danbooru", query, (word,))
    if result:
        return ("danbooru", result[0], result[1])
    
    return None


async def get_translation(word):
    """从数据库中获取单词的翻译，使用LRU缓存"""
    cached = _get_translation_cache(word)
    if cached:
        if cached[0] == "tag_tags":
            return {"translate": cached[1], "color": cached[2], "color_id": -1}
        else:
            return {"translate": cached[1], "color": None, "color_id": cached[2]}
    return None


def clear_translation_cache():
    """清除翻译缓存（在数据更新时调用）"""
    _get_translation_cache.cache_clear()
```

**前端缓存实现**: `src/utils/translation-cache.ts`

```typescript
// src/utils/translation-cache.ts
export class TranslationLRUCache {
  private maxSize: number
  private cache: Map<string, { translation: string; timestamp: number }>

  constructor(maxSize = 200) {
    this.maxSize = maxSize
    this.cache = new Map()
  }

  get(key: string): string | null {
    const item = this.cache.get(key)
    if (item) {
      // 检查是否过期（24小时）
      if (Date.now() - item.timestamp < 24 * 60 * 60 * 1000) {
        // 移到最后表示最近使用
        this.cache.delete(key)
        this.cache.set(key, item)
        return item.translation
      }
      this.cache.delete(key)
    }
    return null
  }

  set(key: string, translation: string): void {
    if (this.cache.has(key)) {
      this.cache.delete(key)
    } else if (this.cache.size >= this.maxSize) {
      // 删除最久未使用的
      const firstKey = this.cache.keys().next().value
      this.cache.delete(firstKey)
    }
    this.cache.set(key, { translation, timestamp: Date.now() })
  }

  clear(): void {
    this.cache.clear()
  }

  getStats() {
    return {
      size: this.cache.size,
      maxSize: this.maxSize
    }
  }
}

export const translationCache = new TranslationLRUCache(200)
```

### 2.2 自动补全搜索算法优化 (P1)

#### 当前实现分析
- 已实现基于匹配分数的排序
- 未使用Trie树/前缀树数据结构
- 大量数据时性能可进一步提升

#### 优化方案：引入Trie树

**文件位置**: `app/server/fast_autocomplete/trie.py`

```python
# app/server/fast_autocomplete/trie.py
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False
        self.data = None  # 存储完整数据
        self.score = 0    # 匹配分数


class Trie:
    def __init__(self):
        self.root = TrieNode()
    
    def insert(self, word: str, data: dict, score: int = 0):
        """插入单词到Trie树"""
        node = self.root
        for char in word.lower():
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.is_end = True
        node.data = data
        node.score = score
    
    def search_prefix(self, prefix: str, limit: int = 10) -> list:
        """搜索前缀匹配的所有单词"""
        node = self.root
        for char in prefix.lower():
            if char not in node.children:
                return []
            node = node.children[char]
        
        # 收集所有匹配结果
        results = []
        self._collect_all(node, prefix, results)
        
        # 按分数排序，返回前limit个
        results.sort(key=lambda x: x['score'], reverse=True)
        return results[:limit]
    
    def _collect_all(self, node: TrieNode, prefix: str, results: list):
        """递归收集所有匹配结果"""
        if node.is_end and node.data:
            results.append({
                **node.data,
                'score': node.score
            })
        
        for char, child in node.children.items():
            self._collect_all(child, prefix + char, results)


# 全局Trie树实例
_trie_instance = None

def get_trie() -> Trie:
    """获取Trie树单例"""
    global _trie_instance
    if _trie_instance is None:
        _trie_instance = Trie()
        # 初始化时加载数据
        # load_data_to_trie(_trie_instance)
    return _trie_instance
```

**优化后的自动补全**: `app/server/fast_autocomplete/autocomplete.py`

```python
from .trie import get_trie

async def fuzzy_search_optimized(query: str, limit: int = 10) -> list:
    """使用Trie树优化的模糊搜索"""
    trie = get_trie()
    
    # 1. 先从Trie树获取前缀匹配结果
    prefix_results = trie.search_prefix(query, limit * 2)
    
    # 2. 如果结果不足，回退到数据库查询
    if len(prefix_results) < limit:
        remaining = limit - len(prefix_results)
        db_results = await fuzzy_search(query, remaining)
        
        # 合并结果，去重
        existing_texts = {r['text'] for r in prefix_results}
        for r in db_results:
            if r['text'] not in existing_texts:
                prefix_results.append(r)
    
    # 3. 按分数排序返回
    prefix_results.sort(key=lambda x: x.get('match_score', 0), reverse=True)
    return prefix_results[:limit]
```

### 2.6 Service Worker离线缓存 (P2)

#### 实现方案

**文件位置**: `src/sw.js`

```javascript
// src/sw.js
const CACHE_NAME = 'weilin-comfyui-tools-v1'
const STATIC_ASSETS = [
  '/',
  '/javascript/main.entry.js',
  '/style.css',
  '/fonts/',
  '/images/'
]

// 安装事件 - 缓存静态资源
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => {
      return cache.addAll(STATIC_ASSETS)
    })
  )
  self.skipWaiting()
})

// 激活事件 - 清理旧缓存
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then((cacheNames) => {
      return Promise.all(
        cacheNames
          .filter((name) => name !== CACHE_NAME)
          .map((name) => caches.delete(name))
      )
    })
  )
  self.clients.claim()
})

// 请求拦截 - 缓存优先策略
self.addEventListener('fetch', (event) => {
  const { request } = event
  
  // 只缓存GET请求
  if (request.method !== 'GET') return
  
  // API请求使用网络优先策略
  if (request.url.includes('/api/')) {
    event.respondWith(networkFirst(request))
    return
  }
  
  // 静态资源使用缓存优先策略
  event.respondWith(cacheFirst(request))
})

// 缓存优先策略
async function cacheFirst(request) {
  const cached = await caches.match(request)
  if (cached) return cached
  
  try {
    const response = await fetch(request)
    const cache = await caches.open(CACHE_NAME)
    cache.put(request, response.clone())
    return response
  } catch (error) {
    return new Response('Network error', { status: 408 })
  }
}

// 网络优先策略
async function networkFirst(request) {
  try {
    const response = await fetch(request)
    const cache = await caches.open(CACHE_NAME)
    cache.put(request, response.clone())
    return response
  } catch (error) {
    const cached = await caches.match(request)
    if (cached) return cached
    return new Response('Network error', { status: 408 })
  }
}
```

```javascript
// src/directives/lazyload.js
export default {
  mounted(el, binding) {
    // 使用IntersectionObserver实现懒加载
    const observer = new IntersectionObserver(
      (entries) => {
        entries.forEach((entry) => {
          if (entry.isIntersecting) {
            const img = entry.target
            const src = img.dataset.src || binding.value
            
            if (src) {
              img.src = src
              img.removeAttribute('data-src')
            }
            
            observer.unobserve(img)
          }
        })
      },
      {
        rootMargin: '50px 0px', // 提前50px开始加载
        threshold: 0.01
      }
    )
    
    observer.observe(el)
    el._lazyObserver = observer
  },
  
  unmounted(el) {
    if (el._lazyObserver) {
      el._lazyObserver.disconnect()
      delete el._lazyObserver
    }
  }
}
```

### 2.8 动态防抖优化 (P0)

#### 实现方案

**文件位置**: `src/utils/dynamic-debounce.ts`

```typescript
// src/utils/dynamic-debounce.ts
export interface DebounceOptions {
  minDelay: number
  maxDelay: number
  leading?: boolean
  trailing?: boolean
}

export function createDynamicDebounce(
  fn: Function,
  options: DebounceOptions = { minDelay: 100, maxDelay: 300 }
) {
  let timer: ReturnType<typeof setTimeout> | null = null
  let lastCallTime = 0
  let lastInputTime = 0

  const getDelay = (): number => {
    const now = Date.now()
    const timeSinceLastInput = now - lastInputTime
    lastInputTime = now

    // 快速输入时使用较长延迟
    if (timeSinceLastInput < 100) {
      return options.maxDelay
    }
    // 正常输入使用中等延迟
    if (timeSinceLastInput < 300) {
      return (options.minDelay + options.maxDelay) / 2
    }
    // 慢速输入使用较短延迟
    return options.minDelay
  }

  return function (this: any, ...args: any[]) {
    const delay = getDelay()

    if (timer) {
      clearTimeout(timer)
    }

    // leading选项：第一次调用立即执行
    if (options.leading && Date.now() - lastCallTime > delay) {
      fn.apply(this, args)
      lastCallTime = Date.now()
      return
    }

    timer = setTimeout(() => {
      fn.apply(this, args)
      lastCallTime = Date.now()
      timer = null
    }, delay)
  }
}

// 使用示例
export const debouncedProcessInput = createDynamicDebounce(
  processInput,
  { minDelay: 100, maxDelay: 200 }
)
```

```javascript
export const performanceMonitor = {
  marks: new Map<string, number>(),

  },

      console.log(`[Performance] ${label}: ${duration.toFixed(2)}ms`)
      this.marks.delete(label)
      return duration
    }
    return 0
  },

    const start = this.marks.get(startMark)
    const end = this.marks.get(endMark)
    if (start && end) {
      const duration = end - start
      console.log(`[Performance] ${label}: ${duration.toFixed(2)}ms`)
      return duration
    }
    return 0
  },
```
